Konva.js

Konva 官網

範例程式碼


    <script>
    var stage = new Konva.Stage({
        container: 'container',
        width: 600,
        height: 400,
    });
    var layer = new Konva.Layer();
    var star = new Konva.Star({
        x: stage.width() / 2,
        y: stage.height() / 2,
        numPoints: 6,
        innerRadius: 40,  // 調整內圓半徑
        outerRadius: 70,  // 調整外圓半徑
        fill: 'yellow',
        stroke: 'black',
        strokeWidth: 4,
    });
    layer.add(star);
    stage.add(layer);
    </script>
