AOS 動畫效果

AOS 官網

範例程式碼

            <!-- AOS CSS 和 JS -->
<link href="https://cdnjs.cloudflare.com/ajax/libs/aos/2.3.4/aos.css" rel="stylesheet">
<script src="https://cdnjs.cloudflare.com/ajax/libs/aos/2.3.4/aos.js"></script>

<!-- 初始化 AOS -->
<script>
    AOS.init();
</script>
            <div data-aos="fade-up">Fade Up 動畫</div>
            <div data-aos="fade-down">Fade Down 動畫</div>
            <div data-aos="fade-left">Fade Left 動畫</div>
            <div data-aos="fade-right">Fade Right 動畫</div>
            <div data-aos="zoom-in">Zoom In 動畫</div>
            <div data-aos="zoom-out">Zoom Out 動畫</div>
            <div data-aos="flip-left">Flip Left 動畫</div>
            <div data-aos="flip-right">Flip Right 動畫</div>
            <div data-aos="slide-up">Slide Up 動畫</div>
            <div data-aos="slide-down">Slide Down 動畫</div>
            <div data-aos="slide-left">Slide Left 動畫</div>
            <div data-aos="slide-right">Slide Right 動畫</div>
