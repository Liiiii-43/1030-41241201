Lottie 動畫

Lottie.js 官網

範例程式碼

<!-- Lottie.js -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/lottie-web/5.7.13/lottie.min.js"></script>

<!-- Lottie 動畫初始化 -->
<script>
    lottie.loadAnimation({
        container: document.getElementById('lottie-animation'), // 容器
        renderer: 'svg', // 渲染方式
        loop: true, // 是否循環播放
        autoplay: true, // 自動播放
        path: 'https://lottie.host/embed/925421de-349c-4b13-8b27-41dc35baea1f/AtOaB6oT1m.json' // JSON 動畫檔案路徑
    });
</script>
